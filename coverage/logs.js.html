export function generateLogs(type, player1, player2, value, specialAttackName = '') {
  const time = new Date().toLocaleTimeString();
  let text = '';
 
  switch (type) {
  case 'start':
    text = logs.start
      .replace('[time]', time)
      .replace('[player1]', player1.name)
      .replace('[player2]', player2.name);
    break;
  case 'hit':
    text = logs.hit[getRandom(logs.hit.length - 1)]
      .replace('[playerDefence]', player2.name)
      .replace('[playerKick]', player1.name);
    text = `${time} ${text} [-${value}] [${player2.hp}/100]`;
    break;
  case 'defence':
    text = logs.defence[getRandom(logs.defence.length - 1)]
      .replace('[playerKick]', player1.name)
      .replace('[playerDefence]', player2.name);
    text = `${time} ${text}`;
    break;
  case 'end':
    text = logs.end[getRandom(logs.end.length - 1)]
      .replace('[playerWins]', player1.name)
      .replace('[playerLose]', player2.name);
    break;
  case 'draw':
    text = logs.draw;
    break;
  case 'special':
    text = logs.special[getRandom(logs.special.length - 1)]
      .replace('[playerKick]', player1.name)
      .replace('[playerDefence]', player2.name)
      .replace('[specialAttackName]', specialAttackName)
      .replace('[value]', value);
    text = `${time} ${text} [-${value}] [${player2.hp}/100]`;
    break;
  case 'critical':
    text = logs.critical[getRandom(logs.critical.length - 1)]
      .replace('[playerKick]', player1.name)
      .replace('[playerDefence]', player2.name)
      .replace('[value]', value);
    text = `${time} ${text} [-${value}] [${player2.hp}/100]`;
    break;
  case 'rage':
    text = logs.rage[getRandom(logs.rage.length - 1)]
      .replace('[playerKick]', player1.name)
      .replace('[playerDefence]', player2.name)
      .replace('[value]', value);
    text = `${time} ${text} [-${value}] [${player2.hp}/100]`;
    break;
  }
 
  const chat = document.querySelector('.chat');
  const el = `<p>${text}</p>`;
  chat.insertAdjacentHTML('afterbegin', el);
}
 
export default generateLogs;
